import os
def move_txt_files(directory, target_folder):
if not os.path.exists(target_folder):
os.makedirs(target_folder)
for file in os.listdir(directory):
if file.endswith(".txt"):
old_path = os.path.join(directory, file)
new_path = os.path.join(target_folder, file)
try:
os.rename(old_path, new_path)
print(f"Moved: {old_path} -> {new_path}")
except Exception as e:
print(f"Error moving {old_path}: {e}")

directory = "/Users/yyuhhhh13/Downloads/tables" #
target_folder = os.path.join(directory, "txt_files") #
